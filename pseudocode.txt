STORE "TOTAL" AS INTEGER WITH ANY VALUE
STORE "HARGA" AS INTEGER WITH ANY VALUE
STORE "KUANTITAS" AS INTEGER WITH ANY VALUE
STORE "DISKON" AS INTEGER WITH ANY VALUE
STORE "BAYAR" AS INTEGER WITH O

STORE "TOTAL" WITH "HARGA" TIMES "KUANTITAS"
DISPLAY "TOTAL"

IF "TOTAL" MORE THAN "15.000.000"
    STORE "DISKON" WITH "TOTAL" TIMES 10%
    STORE "BAYAR" WITH "DISKON"
    DISPLAY "BAYAR"

ELSE "TOTAL" LESS THAN "15.000.000"
    STORE "DISKON" WITH "TOTAL"
    STORE "BAYAR" WITH "DISKON"
    DISPLAY "BAYAR"

END IF
